254  /*

265  {"idMachine":1,"couleur":"rouge","etat":false,"actif":true},
266  {"idMachine":2,"couleur":"jaune","etat":false,"actif":true},

273  {"idBoite":2,"couleur":"jaune","etat":false,"actif":true},
274  {"idBoite":3,"couleur":"bleu","etat":false,"actif":true},

277  */

287  if(jsonObject.has("idPoubelle"))
288  {
289  if(actif && etat)

292  }

294  }

304  {
